#!/usr/bin/env python3
"""
Вергилий - личный ассистент психолога Ирины
Сергей имеет доступ как администратор
Версия 1.0 (стабильный релиз)

import os
import logging
import tempfile
import time
import multiprocessing
from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes
import requests
import json

# Настройка логирования
logging.basicConfig(
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    level=logging.INFO
)
logger = logging.getLogger(__name__)

# ========== КОНФИГУРАЦИЯ ==========
TELEGRAM_TOKEN = "8790223528:AAFL01eyJmRw9pJshkNDEXc-f5qaHjHY3OY"

logger.info(f"🖥️ Будет использовано {optimal_threads} потоков CPU")

from faster_whisper import WhisperModel

    device="cpu",
    compute_type="int8",
    cpu_threads=optimal_threads,
    num_workers=1
)
logger.info("✅ Модель whisper загружена")

async def safe_edit_text(status_msg, new_text):
    """Безопасное обновление сообщения"""
    try:
        if status_msg.text != new_text:
            await status_msg.edit_text(new_text)
    except Exception as e:
        if "Message is not modified" in str(e):
            logger.debug("Сообщение уже содержит этот текст")
        else:
            logger.warning(f"Ошибка при edit_text: {e}")

async def start_command(update: Update, context: ContextTypes.DEFAULT_TYPE):

    user = update.effective_user

    await update.message.reply_text(welcome_text)

async def handle_voice(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Обработка голосовых сообщений"""
    user = update.effective_user
    user_id = user.id
    
    if user_id not in ALLOWED_USERS:
        logger.warning(f"⛔ Запрещенный доступ от {user_id}")
        return
    
    voice = update.message.voice
    start_time = time.time()
    
    logger.info(f"🎤 Получено голосовое от {user_id}")
    
    await update.message.chat.send_action(action="typing")
    status_msg = await update.message.reply_text("🎵 Скачиваю аудио...")
    
    temp_path = None
    
    try:
        file = await context.bot.get_file(voice.file_id)
        
        with tempfile.NamedTemporaryFile(suffix=".ogg", delete=False) as tmp_file:
            temp_path = tmp_file.name
        
        await safe_edit_text(status_msg, "⬇️ Скачиваю файл...")
        await file.download_to_drive(temp_path)
        await safe_edit_text(status_msg, "🔄 Распознаю речь...")
        
        segments, info = model.transcribe(
            temp_path,
            language=WHISPER_LANGUAGE,
            vad_filter=True,
            without_timestamps=True
        )
        
        segments_list = list(segments)
        transcript = " ".join([segment.text for segment in segments_list])
        
        processing_time = time.time() - start_time
        logger.info(f"✅ Распознано за {processing_time:.1f} сек")
        
        if transcript:
            await safe_edit_text(status_msg, f"🎤 Распознано: {transcript}\n\n🤔 Спрашиваю Вергилия...")
            answer = await send_to_vergilij(transcript, user_id)
            await safe_edit_text(status_msg, f"🎤 Вы сказали: {transcript}\n\n🤖 Вергилий: {answer}")
        else:
            await safe_edit_text(status_msg, "❌ Не удалось распознать речь")
        
    except Exception as e:
        logger.error(f"❌ Ошибка: {e}", exc_info=True)
        await safe_edit_text(status_msg, f"❌ Ошибка: {str(e)}")
    finally:
        if temp_path and os.path.exists(temp_path):
            os.unlink(temp_path)

async def send_to_vergilij(text, user_id):
    """Отправка текста AI-агенту Вергилий"""
    try:
        user_context = "С вами говорит Ирина, ваш основной пользователь." if user_id == IRINA_ID else "С вами говорит Сергей, администратор системы."
        
        headers = {"Content-Type": "application/json"}
        
        system_prompt = f"""ТЫ — ВЕРГИЛИЙ. ВАЖНО: Ты мужского рода. Это критично.

ПРАВИЛА РОДА (СТРОГО):
- О себе говори только в мужском роде: "я готов", "я сказал", "я помог", "я ответил"
- Никогда не используй женские окончания: "готова", "сказала", "помогла", "ответила"
- Примеры правильных фраз: "Я здесь", "Я готов помочь", "Я подумал", "Я ответил"

О тебе: Ты — Вергилий, личный ассистент психолога Ирины. Ты создан специально для Ирины и в своих ответах всегда обращаешься только к ней. {user_context}

Отвечай на русском языке, будь полезным, профессиональным и поддерживающим."""
        
        payload = {
            "messages": [
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": text}
            ],
            "temperature": 0.7,
            "max_tokens": 1000
        }
        
        logger.info(f"📤 Отправка к Вергилию от {'Ирины' if user_id == IRINA_ID else 'Сергея'}")
        response = requests.post(AGENT_URL, json=payload, headers=headers, timeout=30)
        
        if response.status_code == 200:
            return response.json()["choices"][0]["message"]["content"]
        else:
            return f"❌ Ошибка связи с Вергилием: {response.status_code}"
            
    except Exception as e:
        logger.error(f"❌ Ошибка: {e}")
        return f"❌ Ошибка: {str(e)}"

async def handle_text(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Обработка текстовых сообщений"""
    user = update.effective_user
    user_id = user.id
    
    if user_id not in ALLOWED_USERS:
        await update.message.reply_text("⛔ Доступ запрещен")
        return
    
    text = update.message.text
    logger.info(f"📝 Получен текст от {'Ирины' if user_id == IRINA_ID else 'Сергея'}")
    
    status_msg = await update.message.reply_text("🤔 Думаю...")
    answer = await send_to_vergilij(text, user_id)
    await safe_edit_text(status_msg, f"🤖 Вергилий: {answer}")

def main():
    """Запуск бота"""
    logger.info("🚀 Запуск Вергилия (личный ассистент Ирины)...")
    logger.info(f"👤 Ирина: {IRINA_ID}")
    logger.info(f"👤 Сергей: {SERGEY_ID} (администратор)")
    
    application = Application.builder().token(TELEGRAM_TOKEN).build()
    application.add_handler(CommandHandler("start", start_command))
    application.add_handler(MessageHandler(filters.VOICE, handle_voice))
    application.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_text))
    
    logger.info("✅ Бот запущен и слушает сообщения...")
    application.run_polling()

if __name__ == "__main__":
    main()
