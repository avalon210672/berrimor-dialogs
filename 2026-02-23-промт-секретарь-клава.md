### 4.1. Создание модуля audio (для работы с аудио)

bash

Вставьте код:

python

\"\"\"

Модуль для извлечения аудио из видеофайлов

\"\"\"

import os

import subprocess

import logging

logger = logging.getLogger(\_\_name\_\_)

def extract_audio_from_video(video_path, audio_path=None):

\"\"\"Извлекает аудио из видео файла\"\"\"

if audio_path is None:

audio_path = video_path.rsplit(\'.\', 1)\[0\] + \'.mp3\'

try:

cmd = \[\'ffmpeg\', \'-i\', video_path, \'-q:a\', \'0\', \'-map\', \'a\', \'-y\', audio_path\]

subprocess.run(cmd, check=True, capture_output=True)

logger.info(f\"Аудио извлечено: {audio_path}\")

return audio_path

except Exception as e:

logger.error(f\"Ошибка извлечения аудио: {e}\")

return None

def compress_audio_for_api(audio_path, max_size_mb=20):

\"\"\"Сжимает аудио для отправки в API\"\"\"

if not os.path.exists(audio_path):

return audio_path

size_mb = os.path.getsize(audio_path) / (1024 \* 1024)

if size_mb \<= max_size_mb:

return audio_path

compressed_path = audio_path.rsplit(\'.\', 1)\[0\] + \'\_compressed.mp3\'

try:

cmd = \[\'ffmpeg\', \'-i\', audio_path, \'-b:a\', \'64k\', \'-y\', compressed_path\]

subprocess.run(cmd, check=True, capture_output=True)

return compressed_path

except Exception:

return audio_path

### 4.2. Создание модуля для голосовых сообщений

bash

mkdir -p transkribator_modules/bot/handlers

nano transkribator_modules/bot/handlers/voice_handler.py

Вставьте код:

python

\"\"\"

Обработчик голосовых сообщений

\"\"\"

import os

import logging

import tempfile

from telegram import Update

from telegram.ext import ContextTypes

from telegram.constants import ChatAction

logger = logging.getLogger(\_\_name\_\_)

async def handle_voice(update: Update, context: ContextTypes.DEFAULT_TYPE):

\"\"\"Обработка голосового сообщения\"\"\"

user = update.effective_user

voice = update.message.voice

logger.info(f\"Получено голосовое от {user.id}, длительность: {voice.duration} сек\")

await update.message.chat.send_action(action=ChatAction.TYPING)

status_msg = await update.message.reply_text(\"🎵 Обрабатываю аудио\...\")

try:

file = await context.bot.get_file(voice.file_id)

with tempfile.NamedTemporaryFile(suffix=\".ogg\", delete=False) as tmp_file:

await file.download_to_drive(tmp_file.name)

audio_path = tmp_file.name

await status_msg.edit_text(\"🔄 Конвертирую в текст\...\")

await status_msg.edit_text(

f\"✅ Аудио получено!\\n\\n\"

f\"📊 Длительность: {voice.duration} сек\\n\"

f\"📝 Размер: {voice.file_size / 1024:.1f} КБ\\n\\n\"

f\"⚡ Функция распознавания речи будет добавлена позже.\"

)

except Exception as e:

logger.error(f\"Ошибка: {e}\", exc_info=True)

await status_msg.edit_text(f\"❌ Ошибка при обработке: {str(e)}\")

finally:

if \'audio_path\' in locals() and os.path.exists(audio_path):

os.unlink(audio_path)

bash

*\# Посмотреть ошибки*

docker logs cyberkitty19-transkribator-bot \--tail 50

*\# Пересобрать с очисткой кэша*

docker-compose build \--no-cache bot

docker-compose up -d bot

### 8.2. Если ошибка импорта модуля

bash

*\# Проверить наличие файлов*

ls -la transkribator_modules/bot/handlers/

*\# Создать недостающие файлы (см. раздел 4)*

### 8.3. Если ошибка с PostgreSQL

bash

*\# Пересоздать volume*

docker-compose down -v

docker volume rm transkribator_postgres-data

docker volume create \--name=transkribator_postgres-data

docker-compose up -d

### 8.4. Если ошибка скачивания аудио (404 Not Found)

Отключите локальный Bot API Server в main.py:

- Закомментируйте блок с if USE_LOCAL_BOT_API:

- Пересоберите бота
